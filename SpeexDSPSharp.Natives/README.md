- android-arm: android-arm/native/libspeexdsp.so
- android-arm64: android-arm64/native/libspeexdsp.so
- android-x64: android-x64/native/libspeexdsp.so
- android-x86: android-x86/native/libspeexdsp.so
- linux-arm: linux-arm/native/speexdsp.so
- linux-arm64: linux-arm64/native/speexdsp.so
- linux-x64: linux-x64/native/speexdsp.so
- linux-x86: linux-x86/native/speexdsp.so
- win-arm: win-arm/native/speexdsp.dll
- win-arm64: win-arm64/native/speexdsp.dll
- win-x64: win-x64/native/speexdsp.dll
- win-x86: win-x86/native/speexdsp.dll
